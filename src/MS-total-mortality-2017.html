%%R -w 8 -h 5 -u in -r 200

# Base
q <- ggplot()

# Edit axes scales for ease of positioning annotations
q <- q + scale_x_continuous(limits = c(0, 5)) 
q <- q + scale_y_continuous(limits = c(0, 5))

# Annotations
# 1st line
q <- q + annotate("text", label = "70,237", x = x1, y = y1, size = tlarge, colour = highlight, hjust = 0, vjust = 0)
q <- q + annotate("text", label = "people", x = (x1 + 2.2), y = y1, size = 14, colour = med, hjust = 0, vjust = 0)

# 2nd line 
q <- q + annotate("text", label = "Died from drug overdoses", x = x2, y = y2, size = tmed, colour = med,
                  hjust = 0, vjust = 0)
q <- q + annotate("text", label = "in 2017", x = (x2 + 3.03), y = y2, size = tmed, colour = highlight,
                  alpha = 0.7, hjust = 0, vjust = 0)

# 3rd line
q <- q + annotate("text", label = "Eclipsing U.S. fatalities in the Vietnam,", x = x3, y = y3, 
                  size = tsmall, colour = med, hjust = 0, vjust = 0)
q <- q + annotate("text", label = "Afghanistan, and Iraq Wars", x = x4, y = y4, 
                  size = tsmall, colour = med, hjust = 0, vjust = 0)
q <- q + annotate("text", label = "combined", x = (x4 + 2.7), y = y4, size = tsmall, colour = highlight, 
                  alpha = 0.7, hjust = 0, vjust = 0)
